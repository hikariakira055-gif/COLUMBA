# 1. Sauvegarde du terminal et nettoyage
tput smcup # Écran alternatif
tput civis # Cache le curseur
tput clear

# 2. Définition des raccourcis de style
ROUGE=$(tput setaf 1)
VERT=$(tput setaf 2)
MAGENTA=$(tput setaf 5)
GRAS=$(tput bold)
RESET=$(tput sgr0)

# 3. Dessin de l'interface fixe
tput cup 0 0
echo "${MAGENTA}${GRAS}=== COLUMBA INTERFACE ===${RESET}"
tput cup 1 0
echo "--------------------------"

# Position pour la zone de chat (Ligne 4)
tput cup 4 0
echo "${VERT}[System]${RESET} Connexion prête."

# 4. Boucle d'interaction
while true; do
    # Calcul de la dernière ligne en temps réel
    derniere_ligne=$(($(tput lines) - 1))
    
    # On va tout en bas pour la saisie
    tput cup $derniere_ligne 0
    tput el # On nettoie la ligne précédente
    
    tput cnorm # On réaffiche le curseur pour taper
    echo -n "${ROUGE}Akira>${RESET} "
    read -r msg
    tput civis # On recache le curseur après la saisie
    
    if [ "$msg" = "exit" ]; then
        break
    fi
done

# 5. Nettoyage avant de quitter (CRITIQUE)
tput cnorm # Remet le curseur
tput rmcup # Quitte l'écran alternatif
echo "COLUMBA Fermé."

#!/bin/bash

dst_ip=$(zenity --entry \
  --title="COLUMBA - Init" \
  --text="Adresse IP cible :" \
  --entry-text="127.0.0.1")

# Sécurité si clic sur Annuler
if [ -z "$dst_ip" ]; then
    zenity --info --text="Fermeture de COLUMBA."
    exit 0
fi

# 2. Boucle de Chat
while true; do
    msg=$(zenity --entry \
      --title="COLUMBA - Sender" \
      --text="Entrez votre message :")

    # Si l'utilisateur ferme la fenêtre ou fait Annuler, on quitte la boucle
    if [ $? -ne 0 ] || [ -z "$msg" ]; then
        break
    fi

    # Encodage Hexa propre sans saut de ligne parasite
    hash=$(echo -n "$msg" | xxd -p | tr -d '\n')

    # Injection ICMP
    ping -c 1 -p "$hash" "$dst_ip" > /dev/null

    # Optionnel : Petite notification discrète en bas de ton écran Arch
    zenity --notification --text="Mot '$msg' injecté vers $dst_ip"
done

zenity --info --text="Session COLUMBA terminée."

# SANS fermer la fenêtre principale !
injecter_paquet() {
    # On récupère ce que l'utilisateur a tapé dans la fenêtre
    # YAD envoie les arguments dans l'ordre des champs : $1=IP, $2=Pseudo, $3=Message
    local ip="$1"
    local msg="$3"

    if [ -n "$msg" ]; then
        # Encodage Hexa
        local hash=$(echo -n "$msg" | xxd -p | tr -d '\n')
        
        # Envoi du Ping secret
        ping -c 1 -p "$hash" "$ip" > /dev/null
        
        # Petite notification pour confirmer que ça a marché
        yad --notification --text="[COLUMBA] '$msg' envoyé à $ip !"
    fi
}
export -f injecter_paquet # Indispensable pour que YAD puisse voir la fonction
